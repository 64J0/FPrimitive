open FPrimitive

/// Composible specifications for your domain types:
type NonEmptyString =
  private
  | NonEmptyString of string
  static member create x =
    Spec.def<string>
    |> Spec.notNull "should not be null"
    |> Spec.notEmpty "should not be empty"
    |> Spec.createModel NonEmptyString x

/// ...also available as computation expression.
type NonEmptyList<'a> =
  private
  | NonEmptyList of 'a list
  static member create xs =
    specModel NonEmptyList xs {
      nonEmpty "list should not be empty"
      lengthBetween 1 10 "list length should be between 1-10" }


/// Access controllers for your critical application resources:
let uncontrolled_critical = ignore

let critical =
  Access.func uncontrolled_critical
  |> Access.once
  |> Access.duringHours 9 17
  |> Access.revokable

/// Try to evaluate
Access.eval () critical
/// Revoke when neccessary
Access.revoke

  static member Invariant : spec1:Spec<'T1> * spec2:Spec<'T2> -> Spec<struct ('T1 * 'T2)>
  static member Invariant : spec1:Spec<'T1> * spec2:Spec<'T2> * spec3:Spec<'T3> -> Spec<struct ('T1 * 'T2 * 'T3)>
  static member Merge : spec1:Spec<'T> * spec2:Spec<'T> -> Spec<'T>
  static member Of : unit -> Spec<'T>
  static member Of : name:string -> Spec<'T>

type Spec<'T> =
  private {Tag: string;
           Requirements: ('T -> bool * string) list;
           Cascade: CascadeMode;
           Dependents: Spec<'T> list;}

from FPrimitive

--------------------
type Access<'T,'TResult> =
  private {Capability: 'T -> Result<'TResult,string list>;
           Revokable: (unit -> unit) option;}
val func : f:('a -> 'b) -> Access<'a,'b>
val once : acc:Access<'a,'b> -> Access<'a,'b>
val duringHours : min:int -> max:int -> acc:Access<'a,'b> -> Access<'a,'b>
val revokable : acc:Access<'a,'b> -> Access<'a,'b>
val eval : x:'a -> Access<'a,'b> -> Result<'b,string list>
val revoke : acc:Access<'a,'b> -> unit
